\section{Introduction}
%==============================================================================

\begin{frame}{Introduction -- Contexte}

\begin{itemize}

\item Les variables qualitatives explicatives sont omniprésentes en économie appliquée :
  \begin{itemize}
  \item Économie du travail : genre, niveau de diplôme, catégorie socio-professionnelle.
  \item Économie de l'innovation : secteur d'activité, région, taille d'entreprise.
  \item Économie industrielle : appartenance à un groupe, type de marché.
  \end{itemize}

\item \textbf{Objectif :} comprendre l'interprétation des coefficients des variables qualitatives dans le modèle linéaire et ses extensions.

\end{itemize}

\end{frame}

\begin{frame}{Introduction -- Utilisations}

\begin{enumerate}

\item \textbf{Effets fixes catégoriels} : indicatrices d'appartenance à un groupe.
  \begin{itemize}
  \item Les coefficients s'interprètent comme des \textbf{écarts moyens} par rapport à une modalité de référence.
  \item Ils ne représentent plus des dérivées (qui n'existent pas).
  \end{itemize}

\item \textbf{Approximation de fonctions non linéaires} :
  \begin{itemize}
  \item Découpage d'une variable continue en intervalles.
  \item Estimation d'une relation non paramétrique par morceaux.
  \end{itemize}

\end{enumerate}

\end{frame}

\section{Modèle sans terme constant}
%==============================================================================

\begin{frame}{Sans constante -- Formalisation}

\begin{itemize}

\item Soit une variable qualitative à $p$ modalités. Pour un échantillon de $N$ individus, on définit les ensembles d'indices :
  \[
  G_j = \{i : \text{individu } i \text{ appartient au groupe } j\}, \quad j = 1, \ldots, p
  \]
  avec $\bigcup_{j=1}^{p} G_j = \{1, \ldots, N\}$ (partition).\newline

\item Les variables indicatrices associées sont définies par :
  \[
  D_{ji} = \begin{cases}
      1 & \text{si } i \in G_j \\
      0 & \text{si } i \notin G_j
  \end{cases}, \quad i = 1, \ldots, N
  \]

\end{itemize}

\end{frame}

\begin{frame}{Sans constante -- Propriétés des indicatrices}

Les variables indicatrices vérifient les propriétés suivantes :

\begin{enumerate}

\item \textbf{Idempotence} : $D_{ji}^2 = D_{ji}$ \quad (car $0^2 = 0$ et $1^2 = 1$).\newline

\item \textbf{Exclusivité mutuelle} : $D_{ji} \cdot D_{ki} = 0$ pour tout $j \neq k$.\newline

\item \textbf{Effectif du groupe} : $\displaystyle\sum_{i=1}^{N} D_{ji} = N_j$.\newline

\item \textbf{Fréquence du groupe} : $\displaystyle\frac{1}{N} \sum_{i=1}^{N} D_{ji} = \frac{N_j}{N}$.

\end{enumerate}

\vspace{0.3cm}

La propriété 4 montre que pour les indicatrices, la moyenne arithmétique calcule des pourcentages.

\end{frame}

\begin{frame}{Sans constante -- Modèle linéaire}

\begin{itemize}

\item Le modèle linéaire s'écrit :
  \[
  y_i = \sum_{j=1}^{p} b_j D_{ji} + u_i, \quad i = 1, \ldots, N
  \]
  avec les hypothèses classiques sur les perturbations :
  \[
  \E[u_i] = 0, \quad \E[u_i^2] = \sigma_u^2, \quad \E[u_i u_j] = 0 \text{ pour } i \neq j
  \]\newline

\item L'espérance conditionnelle dans le groupe $j$ est :
  \[
  \boxed{\E[y_i \mid D] = b_j \quad \text{si } i \in G_j}
  \]
  Les coefficients représentent les \textbf{moyennes conditionnelles par groupe}.

\end{itemize}

\end{frame}

\begin{frame}{Sans constante -- Différence entre groupes}

\begin{itemize}

\item La différence entre deux coefficients s'interprète comme la différence des espérances conditionnelles :
  \[
  b_j - b_k = \E[y_i \mid i \in G_j] - \E[y_i \mid i \in G_k]
  \]\newline

\item \textbf{Distinction avec les variables quantitatives} :
  \begin{itemize}
  \item Variables \textbf{quantitatives} : $b_j = \frac{\partial \E[y]}{\partial x_j}$ (dérivée).
  \item Variables \textbf{qualitatives} : $b_j = \E[y \mid \text{groupe } j]$ (moyenne conditionnelle).
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Sans constante -- Écriture matricielle}

\begin{itemize}

\item Pour chaque individu $i$, on définit le vecteur ligne :
  \[
  D_i = (D_{1i}, D_{2i}, \ldots, D_{pi})_{1 \times p}
  \]
  et le vecteur des paramètres :
  \[
  b = \begin{pmatrix} b_1 \\ b_2 \\ \vdots \\ b_p \end{pmatrix}_{p \times 1}
  \]\newline

\item Le modèle individuel s'écrit :
  \[
  y_i = D_i b + u_i, \quad i = 1, \ldots, N
  \]

\end{itemize}

\end{frame}

\begin{frame}{Sans constante -- MCO (1/2)}

\begin{itemize}

\item L'estimateur des MCO est :
  \[
  \hat{b} = \left( \sum_{i=1}^{N} D_i' D_i \right)^{-1} \sum_{i=1}^{N} D_i' y_i
  \]\newline

\item En utilisant l'idempotence et l'exclusivité mutuelle :
  \[
  D_i' D_i = \begin{pmatrix} D_{1i} \\ \vdots \\ D_{pi} \end{pmatrix} (D_{1i}, \ldots, D_{pi}) = \begin{pmatrix}
      D_{1i} & 0 & \cdots & 0 \\
      0 & D_{2i} & \cdots & 0 \\
      \vdots & & \ddots & \vdots \\
      0 & 0 & \cdots & D_{pi}
  \end{pmatrix}
  \]

\end{itemize}

\end{frame}

\begin{frame}{Sans constante -- MCO (2/2)}

\begin{itemize}

\item En sommant sur les individus :
  \[
  \sum_{i=1}^{N} D_i' D_i = \begin{pmatrix}
      N_1 & 0 & \cdots & 0 \\
      0 & N_2 & \cdots & 0 \\
      \vdots & & \ddots & \vdots \\
      0 & 0 & \cdots & N_p
  \end{pmatrix}
  \]
  où $N_j = \sum_{i=1}^{N} D_{ji}$ est l'effectif du groupe $j$.\newline

\item L'estimateur des MCO des coefficients est :
  \[
  \boxed{\hat{b}_j = \frac{1}{N_j} \sum_{i \in G_j} y_i = \bar{y}_j}
  \]
  C'est la \textbf{moyenne arithmétique de $y$ dans chaque groupe}.

\end{itemize}

\end{frame}

\begin{frame}{Sans constante -- Démonstration}

Pour le second membre de l'estimateur :
\[
\sum_{i=1}^{N} D_i' y_i = \begin{pmatrix}
    \sum_{i=1}^{N} D_{1i} y_i \\
    \vdots \\
    \sum_{i=1}^{N} D_{ji} y_i \\
    \vdots \\
    \sum_{i=1}^{N} D_{pi} y_i
\end{pmatrix} = \begin{pmatrix}
    \sum_{i \in G_1} y_i \\
    \vdots \\
    \sum_{i \in G_j} y_i \\
    \vdots \\
    \sum_{i \in G_p} y_i
\end{pmatrix}
\]

En combinant avec l'inverse de la matrice diagonale :
\[
\hat{b} = \begin{pmatrix}
    1/N_1 & & 0 \\
    & \ddots & \\
    0 & & 1/N_p
\end{pmatrix} \begin{pmatrix}
    \sum_{i \in G_1} y_i \\
    \vdots \\
    \sum_{i \in G_p} y_i
\end{pmatrix} = \begin{pmatrix}
    \bar{y}_1 \\
    \vdots \\
    \bar{y}_p
\end{pmatrix}
\]

\end{frame}

\section{Modèle avec terme constant}
%==============================================================================

\begin{frame}{Avec constante -- Multicolinéarité}

\begin{itemize}

\item Le terme constant $e$ (vecteur unitaire) est égal à la somme des indicatrices :
  \[
  e = \sum_{j=1}^{p} D_j
  \]
  car tout individu appartient exactement à un groupe.\newline

\item Dans un modèle avec terme constant, il faut \textbf{retirer une indicatrice} pour éviter la multicolinéarité parfaite. Cette modalité devient la \textbf{modalité de référence}.

\end{itemize}

\end{frame}

\begin{frame}{Avec constante -- Modalité de référence}

\begin{itemize}

\item Si l'on retire la modalité $k$, le modèle devient :
  \[
  y_i = c_0 + \sum_{j \neq k} c_j D_{ji} + u_i
  \]\newline

\item Les coefficients $c$ sont reliés aux coefficients $b$ du modèle sans constante par :
  \begin{align*}
      c_0 &= b_k \quad \text{(coefficient de la modalité de référence)} \\
      c_j &= b_j - b_k \quad \text{pour } j \neq k \quad \text{(écart à la référence)}
  \end{align*}

\end{itemize}

\end{frame}

\begin{frame}{Avec constante -- Démonstration}

La prévision du modèle avec constante s'écrit :
\[
\hat{y} = c_0 e + \sum_{j \neq k} c_j D_j
\]

En remplaçant $e = \sum_{j=1}^{p} D_j$ :
\begin{align*}
    \hat{y} &= c_0 \sum_{j=1}^{p} D_j + \sum_{j \neq k} c_j D_j \\
    &= (c_0 + c_1) D_1 + \cdots + c_0 D_k + \cdots + (c_0 + c_p) D_p
\end{align*}

La prévision du modèle sans constante est $\hat{y} = \sum_{j=1}^{p} b_j D_j$.

Par unicité de la décomposition, on identifie :
\[
c_0 + c_j = b_j \text{ pour } j \neq k \quad \text{et} \quad c_0 = b_k
\]

\end{frame}

\begin{frame}{Avec constante -- Interprétation}

\begin{center}
    \small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{ll}
        \toprule
        \textbf{Coefficient} & \textbf{Interprétation} \\
        \midrule
        $c_0$ & Moyenne du groupe de référence $k$ : $\E[y_i \mid i \in G_k]$ \\
        $c_j$ & Écart par rapport à la référence : $\E[y_i \mid i \in G_j] - \E[y_i \mid i \in G_k]$ \\
        \bottomrule
    \end{tabular}
\end{center}

\vspace{0.5cm}

Il est \textbf{indispensable} d'indiquer explicitement la modalité de référence retirée dans les tableaux de régression pour permettre une interprétation correcte des résultats.

\end{frame}

\begin{frame}{Avec constante -- Tests statistiques}

\begin{itemize}

\item \textbf{Test de Fisher.} Le test de Fisher sur le modèle avec terme constant teste l'hypothèse :
  \[
  H_0 : c_1 = c_2 = \cdots = c_{k-1} = c_{k+1} = \cdots = c_p = 0
  \]
  Ce qui équivaut à :
  \[
  H_0 : \E[y_i \mid i \in G_j] = \E[y_i \mid i \in G_k] \quad \forall j \neq k
  \]
  C'est un \textbf{test d'égalité des moyennes entre tous les groupes}.\newline

\item \textbf{Test de Student.} Un test de Student sur $c_j$ teste l'égalité des moyennes entre le groupe $j$ et le groupe de référence $k$.

\end{itemize}

\end{frame}

\section{Extensions et interactions}
%==============================================================================

\begin{frame}{Extensions -- Variables quantitatives}

\begin{itemize}

\item On introduit une matrice de variables explicatives $X_i$ :
  \[
  y_i = X_i a + D_i b + u_i
  \]
  où $X_i$ est de dimension $(1 \times m)$ et $D_i$ de dimension $(1 \times p)$.\newline

\item L'espérance conditionnelle dans le groupe $j$ est :
  \[
  \E[y_i \mid X_i, D_{ji} = 1] = X_i a + b_j
  \]

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Interprétation}

\begin{itemize}

\item La différence entre les groupes $j$ et $k$, \textbf{à $X$ fixé}, est :
  \begin{align*}
      &\E[y_i \mid X_i, D_{ji} = 1] - \E[y_i \mid X_i, D_{ki} = 1] \\
      &\quad = (X_i a + b_j) - (X_i a + b_k) = b_j - b_k
  \end{align*}\newline

\item Les résultats précédents restent valables :
  \begin{itemize}
  \item Le terme constant représente le coefficient de l'indicatrice retirée.
  \item Les autres coefficients mesurent l'écart à cette référence.
  \item Ces écarts sont \textbf{contrôlés pour les autres variables} $X$.
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Évaluation d'une politique}

\begin{itemize}

\item On étudie l'effet d'une mesure d'aide (affectée au hasard) sur une variable de performance $y_i$. L'indicatrice de traitement est :
  \[
  D_i = \begin{cases}
      1 & \text{si l'individu } i \text{ est aidé} \\
      0 & \text{sinon}
  \end{cases}
  \]\newline

\item Pour chaque individu, deux \textbf{résultats potentiels} :
  \begin{itemize}
  \item $y_{0i}$ : performance si l'individu $i$ n'est pas aidé.
  \item $y_{1i}$ : performance si l'individu $i$ est aidé.
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Effet causal}

\begin{itemize}

\item L'effet causal recherché est :
  \[
  \delta = \E[y_{1i} - y_{0i}]
  \]
  C'est la moyenne des variations de performance associées à la mesure.\newline

\item On suppose :
  \begin{align*}
      y_{0i} &= a_0 + X_i c_0 + u_{0i} \\
      y_{1i} &= a_1 + X_i c_1 + u_{1i}
  \end{align*}
  où $X_i$ représente les déterminants de la performance.

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Variable observable}

\begin{itemize}

\item On n'observe que l'un des deux résultats potentiels :
  \[
  y_i = D_i \cdot y_{1i} + (1 - D_i) \cdot y_{0i} = \begin{cases}
      y_{1i} & \text{si } D_i = 1 \\
      y_{0i} & \text{si } D_i = 0
  \end{cases}
  \]\newline

\item En substituant :
  \begin{align*}
      y_i &= D_i(a_1 + X_i c_1 + u_{1i}) + (1-D_i)(a_0 + X_i c_0 + u_{0i}) \\
      &= a_0 + X_i c_0 + D_i \underbrace{(a_1 - a_0)}_{a} + D_i X_i \underbrace{(c_1 - c_0)}_{c} + u_i
  \end{align*}

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Modèle avec produits croisés}

\begin{itemize}

\item Le modèle économétrique s'écrit :
  \[
  \boxed{y_i = a_0 + X_i c_0 + D_i \cdot a + (D_i \otimes X_i) \cdot c + u_i}
  \]
  où :
  \begin{itemize}
  \item $a = a_1 - a_0$ : effet du traitement à $X = 0$.
  \item $c = c_1 - c_0$ : modification des pentes par le traitement.
  \item $u_i = D_i u_{1i} + (1-D_i) u_{0i}$ : perturbation composite.
  \end{itemize}

\item Ce modèle autorise une \textbf{hétérogénéité de l'effet du traitement} selon les caractéristiques $X_i$.

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Produit de Kronecker}

\begin{itemize}

\item Soient $A = (a_{ij})$ une matrice $(m \times n)$ et $B$ une matrice $(p \times q)$. Le \textbf{produit de Kronecker} $A \otimes B$ est la matrice $(mp \times nq)$ :
  \[
  A \otimes B = \begin{pmatrix}
      a_{11} B & a_{12} B & \cdots & a_{1n} B \\
      a_{21} B & a_{22} B & \cdots & a_{2n} B \\
      \vdots & \vdots & \ddots & \vdots \\
      a_{m1} B & a_{m2} B & \cdots & a_{mn} B
  \end{pmatrix}
  \]\newline

\item Avec $D_i = (D_{1i}, \ldots, D_{pi})$ vecteur $(1 \times p)$ et $X_i$ vecteur $(1 \times m)$ :
  \[
  D_i \otimes X_i = (D_{1i} X_i, D_{2i} X_i, \ldots, D_{pi} X_i)_{1 \times pm}
  \]
  Chaque bloc $D_{ji} X_i$ correspond aux variables $X$ \textbf{multipliées par l'indicatrice} du groupe $j$.

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Effet moyen}

\begin{itemize}

\item L'effet du traitement est :
  \[
  \delta = \E[y_{1i} - y_{0i}] = (a_1 - a_0) + \E[X](c_1 - c_0) = a + \E[X] c
  \]
  Un estimateur convergent est $\hat{\delta} = \hat{a} + \bar{X} \hat{c}$.\newline

\item Si les variables $X$ sont \textbf{centrées} avant de calculer les produits croisés (i.e., $\bar{X} = 0$), alors :
  \[
  \boxed{\hat{\delta} = \hat{a}}
  \]
  L'effet moyen est directement donné par le coefficient de l'indicatrice.

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Hétéroscédasticité}

\begin{itemize}

\item La perturbation composite $u_i = D_i u_{1i} + (1-D_i) u_{0i}$ implique :
  \[
  \Var[u_i] = \begin{cases}
      \Var[u_{0i}] & \text{si } D_i = 0 \\
      \Var[u_{1i}] & \text{si } D_i = 1
  \end{cases}
  \]\newline

\item Si $\Var[u_{0i}] \neq \Var[u_{1i}]$, le modèle présente une \textbf{hétéroscédasticité par bloc}. Dans ce cas :
  \begin{itemize}
  \item Les MCO restent convergents mais inefficaces.
  \item Il faut utiliser les \textbf{moindres carrés pondérés} ou les \textbf{écarts-types robustes}.
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Cas polytomique}

\begin{itemize}

\item Avec $p$ groupes et des produits croisés :
  \[
  y_i = D_i b + (X_i \otimes D_i) c + u_i
  \]
  où le terme en $X$ seul est retiré car $\sum_{j=1}^{p} X_i D_{ji} = X_i$.\newline

\item L'espérance conditionnelle dans le groupe $j$ devient :
  \[
  \E[y_i \mid X_i, D_{ji} = 1] = X_i c_j + b_j
  \]
  La différence entre groupes $j$ et $k$ est :
  \[
  \gamma_i = X_i(c_j - c_k) + (b_j - b_k)
  \]
  L'effet varie selon les caractéristiques individuelles $X_i$.

\end{itemize}

\end{frame}

\begin{frame}{Extensions -- Centrage}

\begin{itemize}

\item L'effet moyen est :
  \[
  \bar{\gamma} = \frac{1}{N} \sum_{i=1}^{N} \gamma_i = \bar{X}(c_j - c_k) + (b_j - b_k)
  \]\newline

\item Si l'on centre $X$ avant de calculer les produits croisés ($\bar{X} = 0$), alors :
  \[
  \boxed{\bar{\gamma} = b_j - b_k}
  \]
  La différence de coefficients mesure directement l'écart moyen entre groupes, \textbf{une fois éliminé l'effet des variables de $X$}.

\end{itemize}

\end{frame}

\section{Variables dichotomiques}
%==============================================================================

\begin{frame}{Dichotomiques -- Contexte}

\begin{itemize}

\item Les bases de données microéconomiques contiennent fréquemment :
  \begin{itemize}
  \item Des données tronquées ou censurées.
  \item Des informations connues seulement par intervalle.
  \item Des variables purement qualitatives.
  \end{itemize}

\item \textbf{Exemples :}
  \begin{itemize}
  \item Enquête Innovation (SESSI) : fait d'avoir innové, importance d'un déterminant.
  \item Enquête Emploi (INSEE) : statut d'emploi, heures travaillées (observées seulement si emploi).
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Dichotomiques -- Classification}

\begin{center}
    \begin{tikzpicture}[scale=0.65, every node/.style={transform shape},

\begin{frame}{Dichotomiques -- Définition}

\begin{itemize}

\item Une variable dichotomique ne peut prendre que deux modalités exclusives l'une de l'autre (Oui/Non, Supérieur/Inférieur à un seuil, etc.). Par convention :
  \[
  y_i = \begin{cases}
      1 & \text{si l'événement se produit} \\
      0 & \text{sinon}
  \end{cases}
  \]\newline

\item \textbf{Exemple (Innovation) :}
  \[
  y_i = \begin{cases}
      1 & \text{si l'entreprise } i \text{ a innové} \\
      0 & \text{sinon}
  \end{cases}
  \]

\end{itemize}

\end{frame}

\begin{frame}{Dichotomiques -- Objectif}

\begin{itemize}

\item On cherche à expliquer les déterminants de la \textbf{probabilité} que l'événement se produise :
  \[
  \Prob(y_i = 1 \mid X_i) = \text{?}
  \]
  On cherche les variables qui augmentent ou réduisent cette probabilité.\newline

\item \textbf{Idée :} on va construire un \textbf{modèle latent} (inobservable) qui représente le critère de décision sous-jacent.

\end{itemize}

\end{frame}

%==============================================================================
\section{Le modèle de probabilité linéaire}
%==============================================================================

\begin{frame}{MPL -- Spécification}

\begin{itemize}

\item \textbf{Exemples de variables dépendantes binaires :} participer au marché du travail ($Y_i = 1$) ou non ($Y_i = 0$), acheter un produit, faire défaut sur un crédit, voter pour un candidat...\newline

\item Le \textbf{modèle de probabilité linéaire} (MPL) :
  \[
  Y_i = X_i'\beta + \varepsilon_i \quad \text{avec} \quad Y_i \in \{0, 1\}
  \]
  On a :
  \[
  \E[Y_i \mid X_i] = X_i'\beta = \Prob(Y_i = 1 \mid X_i) \quad \text{(loi de Bernoulli)}
  \]\newline

\item \textbf{Question :} peut-on estimer $\beta$ par MCO comme dans le modèle linéaire standard ?

\end{itemize}

\end{frame}

\begin{frame}{MPL -- Hétéroscédasticité structurelle}

$Y_i \mid X_i$ suit une loi de \textbf{Bernoulli} de paramètre $p_i = X_i'\beta$.

\vspace{0.3cm}

\textbf{Variance conditionnelle :}
\[
\Var[Y_i \mid X_i] = p_i(1-p_i) = X_i'\beta \cdot (1 - X_i'\beta)
\]

\begin{center}
\begin{tikzpicture}[scale=0.9]
\begin{axis}[
    width=10cm, height=5cm,
    xlabel={$p = X'\beta$},
    ylabel={$\Var[Y \mid X] = p(1-p)$},
    xmin=0, xmax=1,
    ymin=0, ymax=0.3,
    grid=major,
    thick
]
\addplot[domain=0:1, samples=100, blue, very thick] {x*(1-x)};
\addplot[dashed, red] coordinates {(0.5,0) (0.5,0.25)};
\node at (axis cs:0.5,0.27) {\small max = 0,25};
\end{axis}
\end{tikzpicture}
\end{center}

La variance \textbf{dépend de $X_i$} : hétéroscédasticité \textbf{inhérente} au modèle.

\end{frame}

\begin{frame}{MPL -- Conséquences de l'hétéroscédasticité}

Sous hétéroscédasticité, l'estimateur MCO reste sans biais et convergent, mais :

\begin{enumerate}

\item \textbf{Inefficacité :} $\hat{\beta}_{\text{MCO}}$ n'est plus BLUE (Best Linear Unbiased Estimator).\newline

\item \textbf{Inférence invalide :} la matrice de variance usuelle $\widehat{\Var}(\hat{\beta}) = \hat{\sigma}^2 (X'X)^{-1}$ est \textbf{incorrecte}. Les tests $t$ et $F$ ne sont plus valides.\newline

\item \textbf{Solution partielle :} utiliser les écarts-types robustes de White (HC) :
  \[
  \widehat{\Var}_{\text{HC}}(\hat{\beta}) = (X'X)^{-1} \left( \sum_{i=1}^n \hat{\varepsilon}_i^2 x_i x_i' \right) (X'X)^{-1}
  \]

\end{enumerate}

\end{frame}

\begin{frame}{MPL -- Prédictions hors de $[0,1]$}

Le modèle linéaire n'impose \textbf{aucune contrainte} sur $\hat{Y}_i = X_i'\hat{\beta}$.

\begin{center}
\begin{tikzpicture}[scale=0.8]
\begin{axis}[
    width=12cm, height=5.5cm,
    xlabel={Variable explicative $X$},
    ylabel={$\Prob(Y=1 \mid X)$},
    xmin=-2, xmax=6,
    ymin=-0.3, ymax=1.3,
    grid=major,
    thick,
    legend pos=south east
]
\addplot[only marks, mark=*, mark size=2pt, blue!70] coordinates {
    (-1,0) (-0.5,0) (0,0) (0.3,0) (0.5,0) (0.8,0) (1,0) (1.2,0) (1.5,1)
    (1.8,0) (2,1) (2.2,1) (2.5,0) (2.8,1) (3,1) (3.2,1) (3.5,1) (3.8,1)
    (4,1) (4.2,1) (4.5,1) (5,1) (5.5,1)
};
\addplot[domain=-2:6, samples=100, red, very thick] {0.1 + 0.2*x};
\fill[red!20, opacity=0.5] (axis cs:-2,-0.3) rectangle (axis cs:-0.5,0);

\end{axis}
\end{tikzpicture}
\end{center}

Des \textbf{probabilités négatives} ou \textbf{supérieures à 1} : absurde !

\end{frame}

\begin{frame}{MPL -- Non-normalité des résidus}

Les résidus ne peuvent prendre que \textbf{deux valeurs} :
\[
\varepsilon_i = Y_i - X_i'\beta =
\begin{cases}
1 - X_i'\beta & \text{si } Y_i = 1 \\[6pt]
-X_i'\beta & \text{si } Y_i = 0
\end{cases}
\]

\begin{center}
\begin{tikzpicture}[scale=0.8]
\begin{axis}[
    width=10cm, height=4.5cm,
    xlabel={Résidu $\varepsilon$},
    ylabel={Fréquence},
    ybar,
    bar width=15pt,
    xmin=-1.2, xmax=1.2,
    ymin=0, ymax=0.8,
    xtick={-0.6, 0.4},
    xticklabels={$-p$, $1-p$},
    ytick={0.4, 0.6},
    yticklabels={$1-p$, $p$},
    grid=major,
    thick,
    title={Distribution des résidus (exemple : $p = X'\beta = 0{,}6$)}
]
\addplot[fill=blue!50] coordinates {(-0.6, 0.4) (0.4, 0.6)};
\end{axis}
\end{tikzpicture}
\end{center}

Distribution \textbf{bimodale}, clairement \textbf{non-gaussienne}. Les tests $t$ et $F$ sont invalides en \textbf{échantillon fini}.

\end{frame}

\begin{frame}{MPL -- Effets marginaux constants}

Dans le MPL, l'effet marginal est \textbf{constant} :
\[
\frac{\partial \Prob(Y=1 \mid X)}{\partial X_k} = \beta_k \quad \forall X
\]

\vspace{0.3cm}

\begin{itemize}

\item Passer de $P = 0{,}01$ à $P = 0{,}06$ : effet faible (événement rare reste rare).\newline

\item Passer de $P = 0{,}47$ à $P = 0{,}52$ : effet fort (basculement de majorité).\newline

\item Passer de $P = 0{,}94$ à $P = 0{,}99$ : effet faible (quasi-certitude confirmée).

\end{itemize}

\vspace{0.3cm}

L'effet d'une variable devrait \textbf{s'atténuer aux extrêmes} (saturation).

\end{frame}

\begin{frame}{MPL -- Comparaison des effets marginaux}

\begin{center}
\begin{tikzpicture}[scale=0.85]

    width=12cm, height=5.5cm,

\textbf{Modèles non-linéaires :} l'effet marginal est maximal au centre ($P \approx 0{,}5$) et tend vers 0 aux extrêmes (effet de \textbf{saturation} réaliste).

\end{frame}

\begin{frame}{MPL -- Récapitulatif}

\begin{center}
    \small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{lcc}
        \toprule
        \textbf{Problème} & \textbf{Conséquence} & \textbf{Gravité} \\
        \midrule
        Hétéroscédasticité structurelle & Inférence invalide (corrigeable HC) & Moyenne \\
        Prédictions hors $[0,1]$ & Probabilités absurdes & \textcolor{darkred}{Élevée} \\
        Résidus non-normaux & Tests invalides (asympt. OK) & Moyenne \\
        Effets marginaux constants & Modèle mal spécifié & \textcolor{darkred}{Élevée} \\
        \bottomrule
    \end{tabular}
\end{center}

\vspace{0.5cm}

Le MPL peut être utilisé comme \textbf{approximation rapide} pour des probabilités proches de 0,5, mais il est \textbf{inadapté} pour une modélisation rigoureuse.

\end{frame}

\section{Modèles à variable latente}
%==============================================================================

\begin{frame}{Variable latente -- Principe}

\begin{itemize}

\item On introduit une variable latente $Y_i^*$ continue :
  \[
  Y_i^* = X_i'\beta + \varepsilon_i, \qquad Y_i = \mathbf{1}(Y_i^* > 0)
  \]\newline

\item Si $\varepsilon_i \sim F$ (fonction de répartition), alors :
  \[
  \Prob(Y_i = 1 \mid X_i) = \Prob(Y_i^* > 0 \mid X_i) = \Prob(\varepsilon_i > -X_i'\beta) = F(X_i'\beta)
  \]\newline

\item \textbf{Choix classiques :}
  \begin{itemize}
  \item $F = \Phi$ (normale standard) $\Rightarrow$ \textbf{Modèle Probit}.
  \item $F = \Lambda$ (logistique) $\Rightarrow$ \textbf{Modèle Logit}.
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Variable latente -- Propriétés}

\begin{enumerate}

\item $F(X'\beta) \in [0,1]$ par construction.\newline

\item Effets marginaux décroissants aux extrêmes : $\frac{\partial P}{\partial X_k} = f(X'\beta) \cdot \beta_k$.\newline

\item Estimation par \textbf{maximum de vraisemblance}.

\end{enumerate}

\end{frame}

\begin{frame}{Variable latente -- Comparaison graphique}

\begin{center}
\begin{tikzpicture}[scale=0.85]
\begin{axis}[
    width=12cm, height=6cm,
    xlabel={$X'\beta$},
    ylabel={$\Prob(Y=1 \mid X)$},
    xmin=-4, xmax=4,
    ymin=-0.2, ymax=1.2,
    grid=major,
    thick,
    legend pos=south east
]
\addplot[domain=-4:4, samples=100, red, very thick] {0.5 + 0.25*x};
\addlegendentry{MPL}
\addplot[domain=-4:4, samples=200, blue, very thick] {1/(1+exp(-1.7*x))};
\addlegendentry{Probit}
\addplot[domain=-4:4, samples=200, green!60!black, very thick, dashed] {1/(1+exp(-x))};
\addlegendentry{Logit}
\addplot[dashed, black, thin] coordinates {(-4,0) (4,0)};
\addplot[dashed, black, thin] coordinates {(-4,1) (4,1)};
\node[red] at (axis cs:-3,-0.1) {\small Impossible};
\node[red] at (axis cs:3,1.1) {\small Impossible};
\end{axis}
\end{tikzpicture}
\end{center}

Les modèles Probit et Logit garantissent $P \in [0,1]$ et capturent la \textbf{saturation} aux extrêmes.

\end{frame}

\begin{frame}{Variable latente -- Résumé}

\textbf{Le modèle de probabilité linéaire (MCO sur $Y$ binaire) :}

\begin{itemize}
\item[\textcolor{darkgreen}{\checkmark}] Simple à estimer et interpréter.
\item[\textcolor{darkgreen}{\checkmark}] Peut servir d'approximation locale.
\item[\textcolor{darkred}{$\times$}] Hétéroscédasticité structurelle.
\item[\textcolor{darkred}{$\times$}] Prédictions hors de $[0,1]$.
\item[\textcolor{darkred}{$\times$}] Effets marginaux constants (irréalistes).
\end{itemize}

\vspace{0.3cm}

\textbf{Solutions préférées :}
\begin{itemize}
\item \textbf{Probit} : $\Prob(Y=1 \mid X) = \Phi(X'\beta)$.
\item \textbf{Logit} : $\Prob(Y=1 \mid X) = \Lambda(X'\beta) = \frac{e^{X'\beta}}{1+e^{X'\beta}}$.
\item Nous pourrions, en principe, considérer d'autres distributions.
\end{itemize}

\vspace{0.3cm}

Estimation par \textbf{maximum de vraisemblance}, propriétés asymptotiques bien établies.

\end{frame}

\section{Variables polytomiques ordonnées et de comptage}
%==============================================================================

\begin{frame}{Polytomiques ordonnées -- Définition}

\begin{itemize}

\item Variable qualitative à plus de deux modalités, ordonnées entre elles.\newline

\item \textbf{Exemples :}
  \begin{itemize}
  \item \textbf{Quantitative discrétisée} : part des produits innovants dans le CA --- $[0\%-10\%]$, $]10\%-30\%]$, $]30\%-70\%]$, $]70\%-100\%]$.
  \item \textbf{Appréciation subjective} : importance de la R\&D comme déterminant --- ``Pas du tout'', ``Un peu'', ``Moyennement'', ``Beaucoup''.
  \end{itemize}

\item Dans les deux cas, les modalités traduisent un \textbf{ordre} qui indique l'intensité de la variable.

\end{itemize}

\end{frame}

\begin{frame}{Polytomiques ordonnées -- Modèle latent}

\begin{itemize}

\item Le modèle latent représente la ``vraie valeur'' de la variable :
  \[
  y_i^* = X_i b + u_i, \quad i = 1, \ldots, N
  \]\newline

\item La variable observable prend $r$ modalités :
  \[
  y_i = \begin{cases}
      1 & \text{si } \alpha_0 < y_i^* \leq \alpha_1 \\
      2 & \text{si } \alpha_1 < y_i^* \leq \alpha_2 \\
      \vdots \\
      r & \text{si } \alpha_{r-1} < y_i^* \leq \alpha_r
  \end{cases}
  \]
  avec par convention $\alpha_0 = -\infty$ et $\alpha_r = +\infty$.

\end{itemize}

\end{frame}

\begin{frame}{Polytomiques ordonnées -- Représentation}

\begin{center}
    \begin{tikzpicture}[scale=0.65, every node/.style={transform shape}]
        \draw[thick, ->] (-0.5,0) -- (10.5,0) node[right] {$y^*$};
        \foreach \x in {0, 2, 5, 8, 10} {
            \draw[thick] (\x,0.1) -- (\x,-0.1);
        }
        \node at (0,-0.5) {\footnotesize $-\infty$};
        \node at (2,-0.5) {\footnotesize $\alpha_1$};
        \node at (5,-0.5) {\footnotesize $\alpha_2$};
        \node at (8,-0.5) {\footnotesize $\alpha_{r-1}$};
        \node at (10,-0.5) {\footnotesize $+\infty$};
        \draw[darkblue, very thick] (0,0.5) -- (2,0.5);
        \node[darkblue] at (1,1) {\small $y=1$};
        \draw[darkgreen, very thick] (2,0.5) -- (5,0.5);
        \node[darkgreen] at (3.5,1) {\small $y=2$};
        \draw[orange, very thick, dashed] (5,0.5) -- (8,0.5);
        \node[orange] at (6.5,1) {\small $\cdots$};
        \draw[darkred, very thick] (8,0.5) -- (10,0.5);
        \node[darkred] at (9,1) {\small $y=r$};
    \end{tikzpicture}
\end{center}

\vspace{0.3cm}

\begin{itemize}

\item \textbf{Seuils connus} : cas des variables quantitatives discrétisées.\newline

\item \textbf{Seuils inconnus} : cas des appréciations subjectives (à estimer).

\end{itemize}

\end{frame}

\begin{frame}{Polytomiques ordonnées -- Probabilités}

\begin{itemize}

\item La probabilité d'observer la modalité $j$ est :
  \begin{align*}
      \Prob(y_i = j) &= \Prob(\alpha_{j-1} < y_i^* \leq \alpha_j) \\
      &= \Prob(y_i^* \leq \alpha_j) - \Prob(y_i^* \leq \alpha_{j-1})
  \end{align*}
  pour $j = 1, \ldots, r$.\newline

\item \textbf{Modèle Probit ordonné.} Si $u_i \sim \mathcal{N}(0, \sigma^2)$ :
  \[
  \Prob(y_i = j) = \Phi\left(\frac{\alpha_j - X_i b}{\sigma}\right) - \Phi\left(\frac{\alpha_{j-1} - X_i b}{\sigma}\right)
  \]
  où $\Phi$ est la fonction de répartition de la loi normale centrée réduite.

\end{itemize}

\end{frame}

\begin{frame}{Comptage -- Définition}

\begin{itemize}

\item Variable prenant ses valeurs dans $\N = \{0, 1, 2, \ldots\}$, représentant le nombre d'occurrences d'un événement.\newline

\item \textbf{Exemple (brevets) :} le nombre de brevets déposés par une entreprise sur une année :
  \begin{itemize}
  \item Valeurs entières positives ou nulles.
  \item Événements relativement rares.
  \item Beaucoup d'entreprises à 0 brevet.
  \end{itemize}

\item Ce n'est pas une variable quantitative classique : elle ne peut pas prendre de valeurs négatives et a une nature discrète.

\end{itemize}

\end{frame}

\begin{frame}{Comptage -- Modèle de Poisson}

\begin{itemize}

\item L'espérance étant toujours strictement positive, on utilise une forme exponentielle :
  \[
  \E[y_i \mid X_i, b] = \exp(X_i b + u_i) > 0
  \]\newline

\item On suppose que $y_i$ suit une \textbf{loi de Poisson} de paramètre $\lambda_i = \exp(X_i b)$ :
  \[
  \Prob(y_i = k) = \frac{\exp(-\lambda_i) \lambda_i^k}{k!}, \quad k = 0, 1, 2, \ldots
  \]

\end{itemize}

\end{frame}

\begin{frame}{Comptage -- Sources d'aléa}

\begin{alertblock}{Double source d'aléa}
\begin{enumerate}
\item \textbf{Erreur sur la moyenne} : $\exp(u_i)$ (incertitude sur l'espérance).
\item \textbf{Tirage de Poisson} : aléa intrinsèque du processus de comptage.
\end{enumerate}
\end{alertblock}

\vspace{0.3cm}

\begin{itemize}

\item \textbf{Modèle de Poisson homogène} : $\Var[\exp(u_i)] = 0$ (pas d'hétérogénéité inobservée).\newline

\item \textbf{Modèle de Poisson hétérogène} : $\Var[\exp(u_i)] > 0$ (hétérogénéité présente).

\end{itemize}

\vspace{0.3cm}

Le modèle de Poisson homogène correspond à une loi de durée exponentielle pour le temps entre événements.

\end{frame}

\section{Récapitulatif}
%==============================================================================

\begin{frame}{Récapitulatif -- Variables explicatives}

\begin{center}
    \small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{lll}
        \toprule
        \textbf{Modèle} & \textbf{Interprétation de $b_j$} & \textbf{Remarque} \\
        \midrule

\item Toujours indiquer la modalité de référence.\newline

\item Le test de Fisher teste l'égalité des moyennes entre groupes.\newline

\item Centrer les variables avant les produits croisés facilite l'interprétation.

\end{itemize}

\end{frame}

\item Définir le \textbf{lien} entre variable latente et variable observée.\newline

\item \textbf{Estimer} par maximum de vraisemblance.

\end{enumerate}
